		foo =  true; // foo is now a boolean

Data types

  • Six data types that are primitives:
    • Boolean
    • Null
    • Undefined
    • Number
    • String
    • Symbol

  • and Object

Boolean


	let foo = true;    // foo is true
		foo = 3 > 2;   // foo is true
		foo = 2 == 12; // foo is false

Null


	let temp = null;
	typeof(temp);      // object
	console.log(temp); // null

Undefined


	let temp;
	typeof(temp);      // undefined
	console.log(temp); // undefined

Number


	let num = 12;      // number
		num = 859;     // number
		num = 8 * 9;   // number
		num = 8 / 0;   // Infinity
		num = -8 / 0;  // -Infinity
		num = 8 / 'a'; // NaN

String


	let name = 'Tim';                // string
		name = "Tim";                // string
		name = "He is called 'Tim'"; // string

Symbol


	let id = Symbol();      // symbol
	let id = Symbol("id");  // symbol

	let id1 = Symbol("id");
	let id2 = Symbol("id");
	id1 === id2;            // false

Object


	let user = new Object(); // object
	let user = {};           // object

	// object
	let user = {
		name: "Tim",
		age: 30
	};

reads a string character by character

  • empty or contains only white space is converted to +0
  • whitespace is ignored (' ', '\t', '\n', '\v', '\f', '\r')
  • unary +/- in the zero position is converted to +/-
  • DecimalDigits is converted to DecimalDigits

	let value = true;
	console.log(typeof value); // boolean

	value = String(value);     // now value is a string "true"
	console.log(typeof value); // string

  • If the argument is true, then the result is "true".
  • If the argument is false, then the result is "false".

Argument Type Result
Number
  • NaN, return the String "NaN"
  • infinity, return the String "Infinity"
  • +0 or −0, return the String "0"
  • If x is any Number value other than −0, then ToNumber(ToString(x)) is exactly the same Number value as x

Arithmetic operators

  • Addition (+)
  • Subtraction (-)
  • Division (/)
  • Multiplication (*)
  • Remainder (%)
  • Exponentiation (**)

Addition (+)


	// Number + Number -> addition
	1 + 2          // 3

	// Boolean + Number -> addition
	true + 1       // 2

	// Boolean + Boolean -> addition
	false + false  // 0

	// Number + String -> concatenation
	5 + 'foo'      // "5foo"

	// String + Boolean -> concatenation
	'foo' + false  // "foofalse"

Subtraction (-)


	5 - 3     // 2
	3 - 5     // -2
	'foo' - 3 // NaN

Division (/)


	1 / 2      // returns 0.5 in JavaScript
	1 / 2      // returns 0 in Java 
	// (neither number is explicitly a floating point number)

	1.0 / 2.0  // returns 0.5 in both JavaScript and Java

	2.0 / 0    // returns Infinity in JavaScript
	2.0 / 0.0  // returns Infinity too
	2.0 / -0.0 // returns -Infinity in JavaScript

Multiplication (*)


	2 * 2               // 4
	-2 * 2              // -4
	Infinity * 0        // NaN
	Infinity * Infinity // Infinity
	'foo' * 2           // NaN

Remainder (%)


	12 % 5   // 2
	-1 % 2   // -1
	1 % -2   // 1
	NaN % 2  // NaN
	1 % 2    // 1
	2 % 3    // 2
	-4 % 2   // -0
	5.5 % 2  // 1.5

Exponentiation (**)


	2 ** 3         // 8
	3 ** 2         // 9
	3 ** 2.5       // 15.588457268119896
	10 ** -1       // 0.1
	NaN ** 2       // NaN

	2 ** 3 ** 2    // 512
	2 ** (3 ** 2)  // 512
	(2 ** 3) ** 2  // 64
	-(2 ** 2)      // -4
	(-2) ** 2      // 4

	// string == object - wtf?

	"0" == "";      // false
